# Function to load data based on file type

def load_data(file_path):

    if file_path.endswith('.parquet'):

        print(f'Loading Parquet data from {file_path}')

        return pd.read_parquet(file_path)

    elif file_path.endswith('.csv'):

        print(f'Loading CSV data from {file_path}')

        return pd.read_csv(file_path)

    else:

        raise ValueError(f'Unsupported file format for {file_path}')



# Load data from multiple CSV parts if necessary

def load_csv_chunks(base_path):

    data_frames = []

    for root, _, files in os.walk(base_path):

        for file in files:

            if file.endswith('.csv'):

                file_path = os.path.join(root, file)

                print(f'Loading CSV chunk: {file_path}')

                df = load_data(file_path)

                data_frames.append(df)

    return pd.concat(data_frames, ignore_index=True)



# Main data loading logic

if os.path.isdir(input_path):
